# steps for environment setup
1) pip install uv
2) uv init (if it is first time running uv in env)
3) pip install pipreqs
4) uv add pipreqs
5) pipreqs .
    pipreqs --ignore .venv,venv --force
    pipreqs . --ignore .venv,venv --force
6) uv add -r requirements.txt

# when using uv, and pipreqs, ignore the venv
pipreqs --ignore .venv,venv
